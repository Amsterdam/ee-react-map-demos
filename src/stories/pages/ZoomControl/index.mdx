### JavaScript

```js
const handleZoomInClick = () => {
  if (mapInstance) {
    mapInstance?.setZoom(mapInstance.getZoom() + 1);
  }
};

const handleZoomOutClick = () => {
  if (mapInstance) {
    mapInstance?.setZoom(mapInstance.getZoom() - 1);
  }
};
```
